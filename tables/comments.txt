CREATE TABLE IF NOT EXISTS `comments` (   `comment_id` int(11) PRIMARY KEY NOT NULL AUTO_INCREMENT,   `text` varchar(140) NOT NULL,   `author_id` int(11) NOT NULL,   `post_id` int(11) NOT NULL, `created_at` DATETIME DEFAULT CURRENT_TIMESTAMP ) ENGINE=InnoDB DEFAULT CHARSET=utf8;
ALTER TABLE `comments` ADD FOREIGN KEY (`author_id`) REFERENCES users(`id`);
ALTER TABLE `comments` ADD FOREIGN KEY (`post_id`) REFERENCES posts(`post_id`);
